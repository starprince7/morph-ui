  const [data, setData] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  const [selectedItem, setSelectedItem] = useState(null);
  
  useEffect(() => {
    const fetchData = async () => {
      try {
        setLoading(true);
        const response = await fetch('/api/get-data?endpoint=' + encodeURIComponent('API_ENDPOINT_HERE'));
        if (!response.ok) {
          throw new Error('Failed to fetch data');
        }
        const result = await response.json();
        const payload = result && result.data !== undefined ? result.data : result;
        setData(payload);
      } catch (err) {
        setError(err.message);
      } finally {
        setLoading(false);
      }
    };
    
    fetchData();
  }, []);
  
  if (loading) {
    return (
      <div className="flex flex-col items-center justify-center p-8 min-h-64 bg-[#10002B]">
        <div className="animate-spin rounded-full h-12 w-12 border-4 border-[#240046]/30 border-t-[#C77DFF] mb-4"></div>
        <span className="text-lg font-medium text-[#C77DFF]">Loading...</span>
      </div>
    );
  }
  
  if (error) {
    return (
      <div className="p-6 bg-red-600/20 border-2 border-red-600/30 rounded-xl">
        <div className="flex items-center">
          <div className="w-8 h-8 bg-red-600 rounded-full flex items-center justify-center mr-3">
            <span className="text-white text-sm font-bold">!</span>
          </div>
          <p className="text-red-300 font-medium">Error: {error}</p>
        </div>
      </div>
    );
  }
  
  return (
    <div className="p-6 bg-[#10002B] min-h-screen">
      <div className="max-w-6xl mx-auto">
        <h1 className="text-3xl md:text-4xl font-normal tracking-tight text-[#E0AAFF] mb-8">
          Product Catalog
        </h1>
        
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
          {data.map((item) => (
            <div 
              key={item.id} 
              className="bg-[#240046] hover:bg-[#3C096C] rounded-xl shadow-md hover:shadow-lg hover:scale-[1.02] transition-all duration-200 p-6 border border-[#C77DFF]/20"
            >
              <h2 className="text-xl font-medium text-[#E0AAFF] mb-2">
                {item.name}
              </h2>
              <p className="text-base text-[#C77DFF] mb-4">
                Price: {'$'}{item.price}
              </p>
              <button 
                onClick={() => setSelectedItem(item)}
                className="bg-[#C77DFF] hover:bg-[#9D4EDD] active:scale-[0.98] text-[#10002B] rounded-full px-6 py-3 font-medium transition-all duration-200 focus:ring-2 focus:ring-[#C77DFF]/40 focus:outline-none shadow-md hover:shadow-lg"
              >
                View Details
              </button>
            </div>
          ))}
        </div>
        
        {selectedItem && (
          <div className="fixed inset-0 bg-[#10002B]/80 backdrop-blur-sm flex items-center justify-center p-4 z-50">
            <div className="bg-[#240046] rounded-2xl shadow-2xl p-8 max-w-md w-full transform scale-100 transition-transform border border-[#C77DFF]/30">
              <h3 className="text-2xl font-medium text-[#E0AAFF] mb-4">
                {selectedItem.name}
              </h3>
              <p className="text-lg text-[#C77DFF] mb-6">
                Price: {'$'}{selectedItem.price}
              </p>
              <button 
                onClick={() => setSelectedItem(null)}
                className="border-2 border-[#C77DFF] text-[#C77DFF] hover:bg-[#C77DFF] hover:text-[#10002B] rounded-full px-6 py-3 font-medium transition-all duration-200 focus:ring-2 focus:ring-[#C77DFF]/40 focus:outline-none"
              >
                Close
              </button>
            </div>
          </div>
        )}
      </div>
    </div>
  );
}
